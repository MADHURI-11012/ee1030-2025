\begin{frame}{Question}
A product is made by mixing three raw materials 1, 2, 3 in varying proportions, where material 1 must account for not more than 50\% of the total. If $x, y$ and $z$ are the amounts of raw materials 1, 2, 3, respectively, this constraint can be modeled as \hfill(2007 PI)

\begin{enumerate}[label=\alph*)]
    \item $x \le 0.5$
    \item $x \le 0.5(x + y + z)$
    \item $0.5x \le x + y + z$
    \item $x \ge 0.5(y + z)$
\end{enumerate}
\end{frame}

\begin{frame}{solution}
Given,\\
Raw Materials: 1,2,3\\
Amount: x,y,z\\
let,\\
Total amount of Product: T\\
\begin{align}
    T = x+y+z
\end{align}
\end{frame}
\begin{frame}
given constraint: Material one (x) must not account for more than 50\% of the total.\\
\begin{align} 
    x \le 0.5(T)\\
    x \le 0.5(x+y+z)\\
    x\le y+z
\end{align}

Correct Option: b
\end{frame}
